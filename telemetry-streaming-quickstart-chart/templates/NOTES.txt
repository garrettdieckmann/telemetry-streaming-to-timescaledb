OpenTelemetry Metrics can now be sent to:
export NODE_IP=$(kubectl get nodes --namespace {{ .Values.namespace }} -o jsonpath="{.items[0].status.addresses[0].address}")
export OTEL_COLLECTOR_PORT=$(kubectl get --namespace {{ .Values.namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services otel-collector-ingest-svc)

echo http://$NODE_IP:$OTEL_COLLECTOR_PORT

export GRAFANA_PORT=$(kubectl get --namespace {{ .Values.namespace }} -o jsonpath="{.spec.ports[0].nodePort}" services grafana-external-svc)
